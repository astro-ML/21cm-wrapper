103    def prior_emcee(self, parameters: dict):
104        """Compute the prior probability using the emcee sampler.
105
106        Args:
107            parameters (dict): The parameters.
108
109        Returns:
110            int: The prior probability.
111        """
112        dict1 = parameters
113        dict2 = self.prior_ranges
114
115        def compare_values(val1, range_tuple):
116            return range_tuple[0] <= val1 <= range_tuple[1]
117
118        def compare_nested(dict1, dict2):
119            for key, val in dict1.items():
120                if key not in dict2:
121                    raise KeyMismatchError(
122                        f"Key '{key}' found in dict1 but not in dict2"
123                    )
124                if isinstance(val, dict):
125                    if not isinstance(dict2[key], dict):
126                        raise KeyMismatchError(
127                            f"Value for key '{key}' is a dict in dict1 but not in dict2"
128                        )
129                    if not compare_nested(val, dict2[key]):
130                        return 0
131                else:
132                    if not compare_values(val, dict2[key]):
133                        return 0
134            return 1
135
136        result = compare_nested(dict1, dict2)
137        self.debug(f"Prior is {result}")
138        return result

424        """
425        Replace all values in the nested dictionary with the elements of the values_array in order.
426
427        Args:
428            nested_dict: A dictionary which may contain other dictionaries as values.
429            values_array: A list of values to replace in the nested dictionary.
430
431        Returns:
432            A new dictionary with replaced values.
433
434        Raises:
435            ValueError: If the number of values in the dictionary does not match the length of values_array.
436        """
437        values_array = list(values_array)
438
439        def count_values(d):
440            """Recursively count the total number of values in the nested dictionary."""
441            count = 0
442            for value in d.values():
443                if isinstance(value, dict):
444                    count += count_values(value)
445                else:
446                    count += 1
447            return count
448
449        def replace(d, values):
450            """Recursively replace values in the nested dictionary with values from the list."""
451            for key in d.keys():
452                if isinstance(d[key], dict):
453                    replace(d[key], values)
454                else:
455                    d[key] = values.pop(0)
456
457        total_values = count_values(nested_dict)
458
459        if total_values != len(values_array):
460            raise ValueError(
461                "The number of values in the dictionary does not match the length of the values array."
462            )
463        replace(nested_dict, values_array)
464
465        return nested_dict
